VAR_GLOBAL
 START AT %0000 : BOOL := 0;
END_VAR

VAR_GLOBAL
 STOP AT %0001 : BOOL := 0;
END_VAR

VAR_GLOBAL 
 ALARM AT %0002 : BOOL := 0;
END_VAR

VAR_GLOBAL 
 SILNIK AT %0008  : BOOL := 0;
END_VAR

VAR_GLOBAL 
 POMPA AT %0009 : BOOL := 0;
END_VAR

PROGRAM MAIN
 VAR_EXTERNAL
 START        : BOOL;
 STOP         : BOOL;
 ALARM        : BOOL;
 SILNIK       : BOOL;
 POMPA        : BOOL;
 END_VAR
 
 VAR
    DELAY_ON  : TON;  (* Instnacja bloku TON *)
    DELAY_OFF : TOF;
 END_VAR
 
      SILNIK := (START OR SILNIK) AND NOT STOP AND NOT ALARM;
      DELAY_ON(IN:=SILNIK, PT:=t#5s);
      DELAY_OFF(IN:=DELAY_ON.Q, PT:=t#5s);
      POMPA := DELAY_OFF.Q;
END_PROGRAM
